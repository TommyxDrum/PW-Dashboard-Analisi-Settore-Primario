package it.floro.dashboard.web;

import it.floro.dashboard.domain.Kpi;
import it.floro.dashboard.domain.SampleRecord;
import it.floro.dashboard.service.KpiService;
import it.floro.dashboard.simulator.DataSimulator;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestParam;

import java.time.LocalDate;
import java.util.List;
import java.util.stream.Collectors;

@Controller
public class DashboardController {
    private List<SampleRecord> cached;
    private final KpiService kpiService = new KpiService();

    private void ensureData(long seed, int days, int fields) {
        if (cached == null) {
            cached = new DataSimulator(seed, LocalDate.of(2024,1,1), days, fields).generate();
        }
    }

    @GetMapping("/")
    public String overview(@RequestParam(defaultValue = "2024-01-01") String from,
                           @RequestParam(defaultValue = "2024-12-31") String to,
                           @RequestParam(required = false) String crop,
                           @RequestParam(required = false) String area,
                           Model model) {
        ensureData(42L, 365, 8);
        List<SampleRecord> filtered = cached.stream().filter(r -> {
            boolean ok = true;
            ok &= !r.date().isBefore(LocalDate.parse(from));
            ok &= !r.date().isAfter(LocalDate.parse(to));
            if (crop!=null && !crop.isBlank()) ok &= crop.equals(r.crop());
            if (area!=null && !area.isBlank()) ok &= area.equals(r.area());
            return ok;
        }).collect(Collectors.toList());

        // Simple aggregates
        double avgYieldHa = filtered.stream().mapToDouble(r -> kpiService.compute(r).yieldPerHa()).average().orElse(Double.NaN);
        double avgEff = filtered.stream().mapToDouble(r -> kpiService.compute(r).waterEfficiencyKgPerM3()).average().orElse(Double.NaN);
        double avgCost = filtered.stream().mapToDouble(r -> kpiService.compute(r).unitCostEurPerT()).average().orElse(Double.NaN);
        double avgMargin = filtered.stream().mapToDouble(r -> kpiService.compute(r).unitMarginEurPerT()).average().orElse(Double.NaN);
        double avgRisk = filtered.stream().mapToDouble(r -> kpiService.compute(r).climateRiskIdx()).average().orElse(Double.NaN);

        model.addAttribute("data", filtered);
        model.addAttribute("avgYieldHa", avgYieldHa);
        model.addAttribute("avgEff", avgEff);
        model.addAttribute("avgCost", avgCost);
        model.addAttribute("avgMargin", avgMargin);
        model.addAttribute("avgRisk", avgRisk);
        model.addAttribute("from", from);
        model.addAttribute("to", to);
        model.addAttribute("crop", crop);
        model.addAttribute("area", area);
        return "index";
    }

    @GetMapping("/risorse")
    public String risorse(Model model) {
        ensureData(42L, 365, 8);
        model.addAttribute("data", cached);
        return "risorse";
    }

    @GetMapping("/rischio")
    public String rischio(Model model) {
        ensureData(42L, 365, 8);
        model.addAttribute("data", cached);
        return "rischio";
    }
}
